    function generateStats(team, league) {
        const seed = team.split('').reduce((acc, char) => acc + char.charCodeAt(0), 0);
        const rand = (min, max, offset = 0) => {
            const val = ((seed + offset) * 9301 + 49297) % 233280;
            return min + (val / 233280) * (max - min);
        };

        const homeAdv = league === 'NFL' ? 2.8 : league === 'NBA' ? 3.5 : league === 'MLB' ? 0.3 : league === 'NCAAF' ? 3.5 : 4.2;
        const paceRange = league === 'NBA' ? [96, 104] : league === 'NCAAB' ? [65, 73] : league === 'MLB' ? [145, 165] : [60, 72];

        return {
            offRating: rand(15, 40, 1),
            defRating: rand(15, 35, 2),
            ppg: rand(18, 35, 3),
            papg: rand(17, 30, 4),
            winPct: rand(0.3, 0.75, 5),
            sos: rand(0.4, 0.65, 6),
            toDiff: rand(-12, 15, 7),
            homeAdv: homeAdv,
            extra1: rand(0.4, 0.7, 8),
            extra2: rand(0.35, 0.65, 9),
            pace: rand(paceRange[0], paceRange[1], 10)
        };
    }

    function predictScore() {
        const league = document.getElementById('league').value;
        const homeTeam = document.getElementById('homeTeam').value;
        const awayTeam = document.getElementById('awayTeam').value;

        if (!homeTeam || !awayTeam) {
            alert('Please select both teams');
            return;
        }
        if (homeTeam === awayTeam) {
            alert('Please select different teams');
            return;
        }

        const h = generateStats(homeTeam, league);
        const a = generateStats(awayTeam, league);

        const paceFactorH = h.pace / ((h.pace + a.pace) / 2);
        const paceFactorA = a.pace / ((h.pace + a.pace) / 2);
        const s1h = (h.offRating - a.defRating) * paceFactorH + h.homeAdv;
        const s1a = (a.offRating - h.defRating) * paceFactorA;

        const sosH = 1 + (h.sos - 0.5) * 0.4;
        const sosA = 1 + (a.sos - 0.5) * 0.4;
        const s2h = (h.ppg * sosH * (1 + h.winPct * 0.2) * paceFactorH) + h.homeAdv;
        const s2a = (a.ppg * sosA * (1 + a.winPct * 0.2) * paceFactorA);

        const s3h = ((h.ppg + (100 - a.papg)) / 2) * paceFactorH + h.homeAdv;
        const s3a = ((a.ppg + (100 - h.papg)) / 2) * paceFactorA;

        const s4h = h.ppg * (1 + h.extra1 * 0.3) * (1 + h.extra2 * 0.2) * sosH * paceFactorH + h.homeAdv - (h.toDiff * 0.2);
        const s4a = a.ppg * (1 + a.extra1 * 0.3) * (1 + a.extra2 * 0.2) * sosA * paceFactorA - (a.toDiff * 0.2);

        const homeScore = Math.round((s1h * 0.3 + s2h * 0.3 + s3h * 0.2 + s4h * 0.2) * 10) / 10;
        const awayScore = Math.round((s1a * 0.3 + s2a * 0.3 + s3a * 0.2 + s4a * 0.2) * 10) / 10;

        const conf = Math.abs(homeScore - awayScore) > 10 ? 'High' : Math.abs(homeScore - awayScore) > 5 ? 'Medium' : 'Low';

        document.getElementById('confidence').textContent = conf + ' Confidence';
